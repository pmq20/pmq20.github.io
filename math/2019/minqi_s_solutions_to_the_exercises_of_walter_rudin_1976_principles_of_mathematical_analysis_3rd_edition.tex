\chapter{The Real and Complex Number Systems}

\section{Exercise 1: Mixed Operations between Rational and Irrational Numbers Produce Irrational Numbers}

If $r$ is rational ($r\ne 0$) and $x$ is irrational, prove that $r+x$ and $rx$ are irrational.

\begin{proof}
Denote $\alpha=r+x, \beta=rx$.

Suppose $\alpha\in\mathbb{Q}$. Then $x=\alpha-r$ is also in $\mathbb{Q}$, because $\mathbb{Q}$ is a field that is closed with respect to additive inverse and addition. This contradicts with the presumption that $x\notin\mathbb{Q}$.

Suppose $\beta\in\mathbb{Q}$. Then $x=\beta/r$ is also in $\mathbb{Q}$, because $r\ne 0$ and $\mathbb{Q}$ is a field that is closed with respect to multiplicative inverse and multiplication. This contradicts with the presumption that $x\notin\mathbb{Q}$.
\end{proof}

\section{Exercise 2: $\sqrt{12}$ is an Irrational Number}

Prove that there is no rational number whose square is $12$.

\begin{lemma}\label{l1}
$\forall n\in \mathbb{Z}, \forall p\in \{2, 3\}$,
\[
n^2\equiv 0 \implies n\equiv 0 \pmod p.
\]
\end{lemma}

\begin{proof}
We check the squares of every element of $\mathbb{Z}/2\mathbb{Z}=\{0,1\}$, the ring of integers modulo $2$.
\[0^2 = 0\]
\[1^2 = 1\]

Thus $n^2=0$ implies $n=0$ in $\mathbb{Z}/2\mathbb{Z}$ by exhaustion.

Similarly we check the squares of every element in $\mathbb{Z}/3\mathbb{Z}=\{0,1,2\}$, the ring of integers modulo $3$.
\[0^2 = 0\]
\[1^2 = 1\]
\[2^2 = 1\]

Thus $n^2=0$ implies $n=0$ in $\mathbb{Z}/3\mathbb{Z}$ by exhaustion.
\end{proof}

\begin{lemma}\label{l2}
There is no rational number whose square is $3$.
\end{lemma}

\begin{proof}
We now show that the equation
\begin{equation}\label{e1}
p^2=3
\end{equation}
is not satisfied by any rational $p$. If there was such a $p$, we could write $p=\frac{m}{n}$ where $m\equiv 0 \pmod{3}$ and $n\equiv 0 \pmod{3}$ are not both true. Let us assume this is done. Then \ref{e1} implies
\begin{equation}\label{e2}
m^2=3n^2,
\end{equation}
This shows that $m^2\equiv 0\pmod{3}$ and we have $m\equiv 0 \pmod{3}$ by Lemma~\ref{l1}. So $m^2\equiv 0\pmod{9}$. It follows that the right side of \ref{e2} is divisible by 9, so that $n^2\equiv 0\pmod{3}$, which implies that $n\equiv 0\pmod{3}$ again by Lemma~\ref{l1}.

The assumption that \ref{e1} holds thus leads to the conclusion that both $m\equiv 0 \pmod{3}$ and $n\equiv 0 \pmod{3}$ are true, contrary to our choice of $m$ and $n$. Hence \ref{e1} is impossible for rational $p$.
\end{proof}

\begin{proof}[Main Proof]
Suppose that there existed $p\in\mathbb{Q}$ such that $p^2=12$. Then there exists $m, n\in\mathbb{Z}$ such that
\[
p=\frac{m}{n} \implies 12=\frac{m^2}{n^2} \implies m^2=2\times 6n^2 \implies m^2\equiv 0 \pmod{2}.
\]
So it follows from Lemma~\ref{l1} that
\[
m\equiv 0 \pmod{2}.
\]

Therefore there exists $k\in\mathbb{Z}$ such that $m=2k$, and
\[
4k^2=12n^2 \implies k^2=3n^2 \implies \left(\frac{k}{n}\right)^2=3.
\]

This contradicts with Lemma~\ref{l2}.
\end{proof}

\chapter{Basic Topology}
\section{Exercise 1: Empty Set is a Subset of Every Set}
Prove that the empty set is a subset of every set.
\begin{proof}
Given a set $X$. It is true that $x\in\varnothing$ implies any statement about $X$, because $x\in\varnothing$ is always false.

In particular, $x\in\varnothing$ implies $x\in X$. Thus $\varnothing\subset X$.
\end{proof}

\section{Exercise 2: Algebraic Numbers are Countable}
A complex number $z$ is said to be algebraic if there are integers $a_0,\dots,a_n$, not all zero, such that\[
a_0z^n+a_1z^{n-1}+\dots+a_{n-1}z+a_n=0.
\]

Prove that the set of all algebraic numbers is countable.
\begin{proof}
Denote $\mathcal{A}$ as the set of all algebraic numbers.

For each $N\in\{2,3,4,\dots\}$, define $\mathcal{A}_N$ to be the set of all $z\in\mathcal{A}$ which satisfy the following property: there exist $n\in\{1,2,3,\dots\}$ and $a_0,\dots,a_n\in\mathbb{Z}$ such that \[
a_0z^n+a_1z^{n-1}+\dots+a_{n-1}z+a_n=0
\]and\[
n+|a_0|+|a_1|+\dots+|a_n|=N.
\]

We claim that\[
\mathcal{A}=\bigcup_{N=2}^\infty \mathcal{A}_N.
\] To prove this, note that each $\mathcal{A}_N\subset \mathcal{A}$, it only suffices to show that $\mathcal{A}\subset\cup_{N=2}^\infty \mathcal{A}_N$. Pick $z\in\mathcal{A}$. Then, by definition, there there exist $n\in\{1,2,3,\dots\}$ and $a_0,\dots,a_n\in\mathbb{Z}$ such that $a_0z^n+a_1z^{n-1}+\dots+a_{n-1}z+a_n=0$. Put $n_0=n+|a_0|+|a_1|+\dots+|a_n|$, then $n_0\geqslant 2$ and $z\in\mathcal{A}_{n_0}$. Thus $z\in\cup_{N=2}^\infty \mathcal{A}_N$.

For each $N\in\{2,3,4,\dots\}$, observe that there are only finite choices of\[
n, a_0, a_1,\dots,a_n
\]such that $n+|a_0|+|a_1|+\dots+|a_n|=N$. Further, for each choice of $n, a_0, a_1,\dots,a_n$, the equation $a_0z^n+a_1z^{n-1}+\dots+a_{n-1}z+a_n=0$ has exactly $n$ solutions of $z\in\mathbb{C}$ as shown by 4.14 of \cite{axler2014linear}. Thus each $\mathcal{A}_N$ is a finite set. By the corollary of Theorem 2.12 of \cite{rudin1976principles}, $\mathcal{A}$ is at most countable.

Finally, since $\mathbb{Q}\subset\mathcal{A}$, $\mathcal{A}$ is infinite. Therefore $\mathcal{A}$ is countable.
\end{proof}

\section{Exercise 3: There Exist Non-Algebraic Real Numbers}
Prove that there exist real numbers which are not algebraic.
\begin{proof}
Suppose all real numbers are algebraic. Then $\mathbb{R}$ would be countable as shown in Exercise 2. This contradicts with the corollary of Theorem 2.43 of \cite{rudin1976principles}.
\end{proof}

\section{Exercise 4: Irrational Real Numbers are Uncountable}
Is the set of all irrational real numbers countable?
\begin{proof}
We claim that the set of all irrational real numbers is uncountable.

By the corollary of Theorem 2.13 of \cite{rudin1976principles}, $\mathbb{Q}$ is countable. Note that $\mathbb{R}$ is the union of $\mathbb{Q}$ and the set of all irrational real numbers. If the set of all irrational real numbers was countable, then, by the corollary of Theorem 2.12 of \cite{rudin1976principles}, $\mathbb{R}$ would be countable, which would contradict with Theorem 2.43 of \cite{rudin1976principles}.
\end{proof}

\section{Exercise 5: A Real Bounded Set with $3$ Limit Points}
Construct a bounded set of real numbers with exactly three limit points.
\begin{proof}
Let\[
\begin{split}
E_0&=\left\{\frac{1}{2},\frac{1}{3},\frac{1}{4},\dots\right\},\\
E_1&=\left\{1+\frac{1}{2},1+\frac{1}{3},1+\frac{1}{4},\dots\right\},\\
E_2&=\left\{2+\frac{1}{2},2+\frac{1}{3},2+\frac{1}{4},\dots\right\},\\
E&=E_0\cup E_1\cup E_2.
\end{split}
\]

We claim that $E$ is a bounded set of real numbers with exactly three limit points: $0,1,2$.

Set $M=3$ and $q=0$, then $d(p,q)<M$ for all $p\in E$. Thus $E$ is bounded.

Pick $p\in\{0,1,2\}$ and a neighborhood $N_r(p)$ of $p$. Let\[
q=p+\frac{1}{\lceil \frac{1}{r}\rceil+1},
\]then $q\ne p$, $q\in N_r(p)$ and $q\in E$.

Pick $p\in \mathbb{R}$ such that $p\not\in\{0,1,2\}$, then there exists a neighborhood of $p$ that contains only finitely many points of $E$. Thus $p$ is not a limit point of $E$ by Theorem 2.20 of \cite{rudin1976principles}.
\end{proof}

\section{Exercise 6: $E$, $E'$ and $\bar E$}\label{Exercise 2.6}
Let $E'$ be the set of all limit points of a set $E$. Let $\bar E=E\cup E'$.
\begin{enumerate}
\item Prove that $E'$ is closed.
\item Prove that $E$ and $\bar E$ have the same limit points.
\item Do $E$ and $E'$ always have the same limit points?
\end{enumerate}
\begin{proof}
\begin{enumerate}
\item Suppose $p$ is a limit point of $E'$. Pick a neighborhood $N_r(p)$ of $p$. Then there exists $q\in N_r(p)$ such that $q\ne p$ and $q\in E'$. Let\[
t=\min\left\{\frac{d(p,q)}{2}, \frac{r-d(p,q)}{2}\right\},
\] then there exists $s\in N_t(q)$ such that $s\ne q$ and $s\in E$. Consequently,\[
\begin{split}
d(s,p)&\leqslant d(s,q)+d(q,p)\\
&<t+d(q,p)\\
&\leqslant\frac{r-d(p,q)}{2}+d(q,p)\\
&=\frac{d(q,p)+r}{2}\\
&<\frac{r+r}{2}\\
&=r.
\end{split}
\]
Hence $s\in N_r(p)$. Moreover,\[
\begin{split}
d(s,p)&\geqslant d(p,q)-d(q,s)\\
&>d(p,q)-t\\
&\geqslant d(p,q)-\frac{d(p,q)}{2}\\
&=\frac{d(p,q)}{2}\\
&>0.
\end{split}
\]
Hnece $s\ne p$. Therefore $p$ is also a limit point of $E$, which implies $p\in E'$.

We have shown above that every limit point of $E'$ is in $E'$. Thus $E'$ is closed.
\item Suppose $p$ is a limit point of $E$. Pick a neighborhood $N_r(p)$. Then there exists $q\in N_r(p)$ such that $q\ne p$ and $q\in E$. Since $E\subset\bar E$, it follows that $q\in \bar E$. Hence $p$ is also a limit point of $\bar E$.

On the other hand, suppose $p$ is a limit point of $\bar E$. Pick a neighborhood $N_r(p)$. Then there exists $q\in N_r(p)$ such that $q\ne p$ and $q\in\bar{E}$. If $q\in E$, let $s=q$. If $q\in E'$, then obtain $s$ using the same notation as (1). Either way, there exists $s\in N_r(p)$ such that $s\in E$ and $s\ne p$. Hence $p$ is also a limit point of $E$.
\item No, $E$ and $E'$ do not always have the same limit points. Suppose\[
E=\left\{\frac{1}{2},\frac{1}{3},\frac{1}{4},\dots\right\}.
\]

Then\[
\begin{split}
E'&=\{0\},\\
E''&=\varnothing.
\end{split}
\]

However, in (1), we have proven that $E''\subset E'$. Hence, in general, $E$ has more limit points than $E'$.
\end{enumerate}
\end{proof}

\chapter{Numerical Sequences and Series}
\section{Exercise 1: Convergence of $\{s_n\}$ Implies Convergence of $\{|s_n|\}$}
Prove that convergence of $\{s_n\}$ implies convergence of $\{|s_n|\}$. Is the converse true?
\begin{proof}
Suppose\[
s_n\to p.
\]

We claim that\[
|s_n|\to |p|.
\]

In order to prove this, pick $\epsilon>0$. Then there exists an integer $N$ such that $n\geqslant N$ implies that $|s_n-p|<\epsilon$. By the reverse triangle inequality,\[
\left||s_n|-|p|\right|\leqslant \left|s_n-p\right|,
\]

Hence $n\geqslant N$ also implies that \[
\left||s_n|-|p|\right|<\epsilon.
\]

Finally, we claim that the converse is not true. Suppose\[
s_n=(-1)^n,
\] then $|s_n|\to 1$ yet $s_n$ diverges.
\end{proof}
