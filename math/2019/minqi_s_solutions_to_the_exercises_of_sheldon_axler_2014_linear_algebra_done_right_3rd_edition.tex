\section{Exercise 1.A.1}
Suppose $a$ and $b$ are real numbers, not both $0$. Find real numbers $c$ and $d$ such that
\[
1/(a+bi)=c+di
\]
\begin{proof}[Answer]
Since at least one of the real numbers $a, b$ is different from $0$, we have $a^2+b^2\ne 0$. We can then let
\[
c=\frac{a}{a^2+b^2},
\]
\[
d=\frac{-b}{a^2+b^2}.
\]
Then
\begin{equation*}
\begin{split}
&\left(a+bi\right)\left(\frac{a}{a^2+b^2}-\frac{b}{a^2+b^2}i\right)\\
&=\left(a\frac{a}{a^2+b^2}-b\left(-\frac{b}{a^2+b^2}\right)\right) + \left(a\left(-\frac{b}{a^2+b^2}\right)+b\frac{a}{a^2+b^2}\right)i\\
&=1 + 0i\\
&=1.
\end{split}
\end{equation*}
\end{proof}

\section{Exercise 1.A.2}
Show that
\[
\frac{-1+\sqrt{3}i}{2}
\]
is a cube root of $1$ (meaning that its cube equals $1$).
\begin{proof}[Answer]
\begin{equation*}
\begin{split}
&\frac{-1+\sqrt{3}i}{2}*\frac{-1+\sqrt{3}i}{2}*\frac{-1+\sqrt{3}i}{2}\\
&=\frac{-2-2\sqrt{3}i}{4}*\frac{-1+\sqrt{3}i}{2}\\
&=\frac{(-2)(-1)-(-2\sqrt{3})(\sqrt{3})+(-2\sqrt{3}+(-2\sqrt{3})(-1))i}{8}\\
&=1
\end{split}
\end{equation*}
\end{proof}

\section{Exercise 1.A.4}
Show that $\alpha+\beta=\beta+\alpha$ for all $\alpha, \beta\in\mathbb{C}$.
\begin{proof}
Suppose $\alpha=a+bi$ and $\beta=c+di$, where $a,b,c,d\in\mathbb{R}$. Then the definition of addition of complex numbers shows that
\[
\alpha+\beta=(a+c)+(b+d)i
\]
and
\[
\beta+\alpha=(c+a)+(d+b)i.
\]
The equations above and the commutativity of addition of real numbers show that $\alpha+\beta=\beta+\alpha$.
\end{proof}

\section{Exercise 1.A.5}
Show that $(\alpha+\beta)+\lambda=\alpha+(\beta+\lambda)$ for all $\alpha,\beta,\lambda\in\mathbb{C}$.
\begin{proof}
Suppose $\alpha=a+bi$, $\beta=c+di$ and $\lambda=e+fi$, where $a,b,c,d,e,f\in\mathbb{R}$. Then the definition of addition of complex numbers shows that
\begin{equation*}
\begin{split}
(\alpha+\beta)+\lambda&=((a+c)+(b+d)i)+e+fi\\
&=((a+c)+e)+((b+d)+f)i
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
\alpha+(\beta+\lambda)&=a+bi+((c+e)+(d+f)i)\\
&=(a+(c+e))+(b+(d+f))i.
\end{split}
\end{equation*}
The equations above and the associativity of addition of real numbers show that $(\alpha+\beta)+\lambda=\alpha+(\beta+\lambda)$.
\end{proof}

\section{Exercise 1.A.6}
Show that $(\alpha\beta)\lambda=\alpha(\beta\lambda)$ for all $\alpha,\beta,\lambda\in\mathbb{C}$.\begin{proof}
Suppose $\alpha=a+bi$, $\beta=c+di$ and $\lambda=e+fi$, where $a,b,c,d,e,f\in\mathbb{R}$. Then the definition of multiplication of complex numbers shows that
\begin{equation*}
\begin{split}
(\alpha\beta)\lambda&=((ac-bd)+(ad+bc)i)(e+fi)\\
&=((ac-bd)e-(ad+bc)f)+((ac-bd)f+(ad+bc)e)i\\
&=(ace-bde-adf-bcf)+(acf-bdf+ade+bce)i
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
\alpha(\beta\lambda)&=(a+bi)((ce-df)+(cf+de)i)\\
&=(a(ce-df)-b(cf+de))+(a(cf+de)+b(ce-df))i\\
&=(ace-adf-bcf-bde)+(acf+ade+bce-bdf)i.
\end{split}
\end{equation*}
The equations above and the commutativity of addition of real numbers show that $(\alpha\beta)\lambda=\alpha(\beta\lambda)$.
\end{proof}

\section{Exercise 1.A.7: Existence and Uniqueness of Additive Inverse in $\mathbb{C}$}
Show that for every $\alpha\in\mathbb{C}$, there exists a unique $\beta\in\mathbb{C}$ such that $\alpha+\beta=0$.
\begin{proof}
Suppose $\alpha=a+bi$ where $a,b\in\mathbb{R}$, then $\beta=-a-bi$ satisfies $\alpha+\beta=0$.

Suppose that there exists another $\beta'\in\mathbb{C}$ such that $\alpha+\beta'=0$. Then
\begin{equation*}
\begin{split}
\beta&=\beta+(\alpha+\beta')\\
&=(\beta+\alpha)+\beta'\\
&=\beta'.
\end{split}
\end{equation*}

The equations above show that $\beta'=\beta$.
\end{proof}

\section{Exercise 1.A.8: Existence and Uniqueness of Multiplicative Inverse in $\mathbb{C}$}
Show that for every $\alpha\in\mathbb{C}$ with $\alpha\ne 0$, there exists a unique $\beta\in\mathbb{C}$ such that $\alpha\beta=1$.
\begin{proof}
The existence of $\beta$ has been shown in Exercise 1.A.1.

Suppose that there exists another $\beta'\in\mathbb{C}$ such that and $\alpha\beta'=1$. Then
\begin{equation*}
\begin{split}
\beta&=\beta(\alpha\beta')\\
&=(\beta\alpha)\beta'\\
&=\beta'.
\end{split}
\end{equation*}

The equations above show that $\beta'=\beta$.
\end{proof}

\section{Exercise 1.A.9}
Show that $\lambda(\alpha+\beta)=\lambda\alpha+\lambda\beta$ for all $\lambda,\alpha,\beta\in\mathbb{C}$.
\begin{proof}
Suppose $\alpha=a+bi$, $\beta=c+di$ and $\lambda=e+fi$, where $a,b,c,d,e,f\in\mathbb{R}$. Then the definition of multiplication and addition of complex numbers shows that
\begin{equation*}
\begin{split}
\lambda(\alpha+\beta)&=(e+fi)((a+c)+(b+d)i)\\
&=(e(a+c)-f(b+d))+(e(b+d)+f(a+c))i\\
&=(ea+ec-fb-fd)+(eb+ed+fa+fc)i
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
\lambda\alpha+\lambda\beta&=(e+fi)(a+bi)+(e+fi)(c+di)\\
&=((ea-fb)+(eb+fa)i)+((ec-fd)+(ed+fc)i)\\
&=(ea-fb+ec-fd)+(eb+fa+ed+fc)i.
\end{split}
\end{equation*}

The equations above and the commutativity of addition of real numbers show that $\lambda(\alpha+\beta)=\lambda\alpha+\lambda\beta$.
\end{proof}

\section{Exercise 1.A.11: A Scalar Multiplication Example in $\mathbb{C}^3$}
Explain why there does not exist $\lambda\in\mathbb{C}$ such that
\[
\lambda(2-3i,5+4i,-6+7i)=(12-5i,7+22i,-32-9i).
\]
\begin{proof}
If there exists such $\lambda$ then the definition of scalar multiplication in $\mathbb{C}^3$ shows that,
\[
\lambda(2-3i)=12-5i.
\]

Thus
\[
\lambda=\frac{12-5i}{2-3i}=3+2i.
\]

But
\begin{equation*}
\begin{split}
\lambda(2-3i,5+4i,-6+7i)&=(3+2i)(2-3i,5+4i,-6+7i)\\
&=(12-5i,7+22i,-32+9i)\\
&\ne(12-5i,7+22i,-32-9i).
\end{split}
\end{equation*}

Therefore $\lambda$ does not exist.
\end{proof}

\section{Exercise 1.A.12: Associativity of Vector Addition in $\mathbb{F}^n$}
Show that $(x+y)+z=x+(y+z)$ for all $x,y,z\in\mathbb{F}^n$.
\begin{proof}
Suppose $x=(x_1,\dots,x_n)$, $y=(y_1,\dots,y_n)$ and $z=(z_1,\dots,z_n)$ where $x_1,\dots,x_n,y_1,\dots,y_n,z_1,\dots,z_n\in\mathbb{F}$. Then the definition of addition in $\mathbb{F}^n$ shows that
\begin{equation*}
\begin{split}
(x+y)+z&=(x_1+y_1,\dots,x_n+y_n)+(z_1,\dots,z_n)\\
&=((x_1+y_1)+z_1,\dots,(x_n+y_n)+z_n)
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
x+(y+z)&=(x_1,\dots,x_n)+(y_1+z_1,\dots,y_n+z_n)\\
&=(x_1+(y_1+z_1),\dots,x_n+(y_n+z_n))
\end{split}
\end{equation*}

The equations above and the associativity of addition in $\mathbb{F}$ show that $(x+y)+z=x+(y+z)$.
\end{proof}

Show that $(ab)x=a(bx)$ for all $x\in\mathbb{F}^n$ and all $a,b\in\mathbb{F}$.

Suppose $x=(x_1,\dots,x_n)$ where $x_1,\dots,x_n\in\mathbb{F}$. Then the definition of scalar multiplication in $\mathbb{F}^n$ shows that

and

\end{split}
\end{equation*}

The equations above and the associativity of multiplication in $\mathbb{F}$ show that $(ab)x=a(bx)$.
\end{proof}

Show that $1x=x$ for all $x\in\mathbb{F}^n$.

\section{Exercise 1.A.15: Distributing Vector Additions with Scalar Multiplications in $\mathbb{F}^n$}
Show that $\lambda(x+y)=\lambda x+\lambda y$ for all $\lambda\in\mathbb{F}$ and $x,y\in\mathbb{F}^n$.
\begin{proof}
Suppose $x=(x_1,\dots,x_n)$ and $y=(y_1,\dots,y_n)$ where $x_1,\dots,x_n,y_1,\dots,y_n\in\mathbb{F}$. Then the definition of addition and scalar multiplication in $\mathbb{F}^n$ shows that
\begin{equation*}
\begin{split}
\lambda(x+y)&=\lambda(x_1+y_1,\dots,x_n+y_n)\\
&=(\lambda(x_1+y_1),\dots,\lambda(x_n+y_n))
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
\lambda x+\lambda y&=(\lambda x_1,\dots,\lambda x_n) + (\lambda y_1,\dots,\lambda y_n)\\
&=(\lambda x_1+\lambda y_1,\dots,\lambda x_n+\lambda y_n)
\end{split}
\end{equation*}

The equations above and the distributive property of $\mathbb{F}$ show that $\lambda(x+y)=\lambda x+\lambda y$.
\end{proof}

Prove that $-(-v)=v$ for every $v\in V$.

\begin{equation*}
\begin{split}
-(-v)&=-(-v)+0\\

\section{Exercise 1.B.2}
Suppose $a\in\mathbb{F}, v\in V$, and $av=0$. Prove that $a=0$ or $v=0$.
\begin{proof}
For the purpose of arriving at a contradiction, assume that both $a\ne 0$ and $v\ne 0$. Then
\[
v=1v=(\frac{1}{a}a)v=\frac{1}{a}(av)=0
\]
which contradicts with $v\ne 0$.

Therefore $a=0$ or $v=0$.
\end{proof}

\section{Exercise 1.B.3}
Suppose $v,w\in V$. Explain why there exists a unique $x\in V$ such that $v+3x=w$.
\begin{proof}
We claim that the following $x\in V$ satisfies $v+3x=w$:
\[
x=\frac{1}{3}(w-v).
\]

We can verify it by showing that,
\begin{equation*}
\begin{split}
v+3x&=v+3\times\frac{1}{3}(w-v)\\
&=v+w-v\\
&=w.
\end{split}
\end{equation*}

Suppose that there exists another $x'\in V$ that also satisfies $v+3x'=w$, then
\begin{equation*}
\begin{split}
0&=w-w\\
&=(v+3x)-(v+3x')\\
&=v+3x-v-3x'\\
&=3x-3x'.
\end{split}
\end{equation*}

Thus
\begin{equation*}
\begin{split}
x'&=x'+\frac{1}{3}\times0\\
&=x'+\frac{1}{3}(3x-3x')\\
&=x'+x-x'\\
&=x.
\end{split}
\end{equation*}
\end{proof}

\section{Exercise 1.B.5}
Show that in the definition of a vector space (1.19), the additive inverse condition can be replaced with the condition that
\[
0v=0, \forall v\in V.
\]

Here the $0$ on the left side is the number $0$, and the $0$ on the right side is the additive identity of $V$. 
\begin{proof}
We will show that the new condition is equal to the original condition, which states that for every $v\in V$, there exists $w\in V$ such that $v+w=0$. That the old condition implies the new condition has been shown in Theorem 1.29 ``The number $0$ times a vector''. Therefore it suffices to show that the new condition implies the old condition.

Suppose that the new condition holds.

Pick $v\in V$, and let $w=(-1)v$. It follows that
\begin{equation*}
\begin{split}
v+w&=1v+(-1)v\\
&=(1-1)v\\
&=0v\\
&=0.
\end{split}
\end{equation*}

Therefore the old condition holds.
\end{proof}

\section{Exercise 1.B.6}
Let $\infty$ and $-\infty$ denote two distinct objects, neither of which is in $\mathbb{R}$. Define an addition and scalar multiplication on $\mathbb{R}\cup\{\infty\}\cup\{-\infty\}$ as you could guess from the notation. Specifically, the sum and product of two real numbers is as usual, and for $t\in\mathbb{R}$ define
\[
\begin{array}{cc}
t\infty=\begin{cases} 
-\infty & \text{if } t<0,\\
0 & \text{if } t=0,\\
\infty & \text{if } t>0,
\end{cases}
&
t(-\infty)=\begin{cases} 
\infty & \text{if } t<0,\\
0 & \text{if } t=0,\\
-\infty & \text{if } t>0,
\end{cases}
\end{array}
\]
\[
\begin{array}{cc}
t+\infty=\infty+t=\infty,
&
t+(-\infty)=(-\infty)+t=(-\infty),
\end{array}
\]
\[
\begin{array}{ccc}
\infty+\infty=\infty,
&
(-\infty)+(-\infty)=(-\infty),
&
\infty+(-\infty)=0.
\end{array}
\]
Is $\mathbb{R}\cup\{\infty\}\cup\{-\infty\}$ a vector space over $\mathbb{R}$? Explain.
\begin{proof}[Answer]
No. Because the associativity of vector addition fails:
\[
(1024+\infty)-\infty=\infty-\infty=0
\]
but
\[
1024+(\infty-\infty)=1024+0=1024.
\]
\end{proof}

\section{Exercise 1.C.1}
For each of the following subsets of $\mathbb{F}^3$, determine whether it is a subspace of $\mathbb{F}^3$:
\begin{enumerate}[label=(\alph*)]
\item $\{(x_1,x_2,x_3)\in\mathbb{F}^3: x_1+2x_2+3x_3=0\}$;
\item $\{(x_1,x_2,x_3)\in\mathbb{F}^3: x_1+2x_2+3x_3=4\}$;
\item $\{(x_1,x_2,x_3)\in\mathbb{F}^3: x_1x_2x_3=0\}$;
\item $\{(x_1,x_2,x_3)\in\mathbb{F}^3: x_1=5x_3\}$.
\end{enumerate}
\begin{proof}[Answer]
\begin{enumerate}[label=(\alph*)]
\item Denote $A=\{(x_1,x_2,x_3)\in\mathbb{F}^3: x_1+2x_2+3x_3=0\}$. We claim that $A$ is a subspace of $\mathbb{F}^3$ with the following verifications.
\begin{enumerate}[label=(\roman*)]
\item $(0,0,0)\in A$ because $0+2*0+3*0=0$.
\item Pick $u,w\in A$, where $u=(u_1,u_2,u_3), w=(w_1,w_2,w_3)$. It follows that $u+w=(u_1+w_1, u_2+w_2, u_3+w_3)$. We can show that $u+w\in A$ by the following equations:
\begin{equation*}
\begin{split}
&(u_1+w_1)+2(u_2+w_2)+3(u_3+w_3)\\
&=(u_1+2u_2+3u_3)+(w_1+2w_2+3w_3)\\
&=0+0=0.
\end{split}
\end{equation*}
\item Pick $a\in \mathbb{F}$ and $u\in A$, where $u=(u_1,u_2,u_3)$. It follows that $au=(au_1,au_2,au_3)$. We can show that $au\in A$ by the following equations:
\begin{equation*}
\begin{split}
(au_1)+2(au_2)+3(au_3)&=a(u_1+2u_2+3u_3)\\
&=a0=0.
\end{split}
\end{equation*}
\end{enumerate}
\item Denote $B=\{(x_1,x_2,x_3)\in\mathbb{F}^3: x_1+2x_2+3x_3=4\}$. $B$ is not a subspace of $\mathbb{F}^3$, because vector $(0,0,0)\notin B$ (as $0+2*0+3*0\ne 4$).
\item Denote $C=\{(x_1,x_2,x_3)\in\mathbb{F}^3: x_1x_2x_3=0\}$. $C$ is not a subspace of $\mathbb{F}^3$ because by setting $u=(0,1,1)\in C$ and $w=(1,1,0)\in C$, we find out that $u+w=(1,2,1) \notin C$.
\item Denote $D=\{(x_1,x_2,x_3)\in\mathbb{F}^3: x_1=5x_3\}$. We claim that $D$ is a subspace of $\mathbb{F}^3$ with the following verifications.
\begin{enumerate}[label=(\roman*)]
\item $(0,0,0)\in A$ because $0=5*0$.
\item Pick $u,w\in A$, where $u=(u_1,u_2,u_3), w=(w_1,w_2,w_3)$. It follows that $u+w=(u_1+w_1, u_2+w_2, u_3+w_3)$. We can show that $u+w\in A$ by the following equations:
\begin{equation*}
\begin{split}
u_1+w_1 &= 5u_3+5w_3\\
&=5(u_1+w_3)
\end{split}
\end{equation*}
\item Pick $a\in \mathbb{F}$ and $u\in A$, where $u=(u_1,u_2,u_3)$. It follows that $au=(au_1,au_2,au_3)$. We can show that $au\in A$ by the following equations:
\begin{equation*}
\begin{split}
au_1&=a(5u_3)\\
&=5(au_3).
\end{split}
\end{equation*}
\end{enumerate}
\end{enumerate}
\end{proof}

\section{Exercise 1.C.2}
Verify all the assertions in Example 1.35.
\begin{enumerate}[label=(\alph*)]
\item If $b\in\mathbb{F}$, then
\[
\{(x_1,x_2,x_3,x_4)\in\mathbb{F}^4: x_3=5x_4+b\}
\]
is a subspace of $\mathbb{F}^4$ if and only if $b=0$.
\item The set of continuous real-values functions on the interval $[0,1]$ is a subspace of $\mathbb{R}^{[0,1]}$.
\item The set of differentiable real-valued functions on $\mathbb{R}$ is a subspace of $\mathbb{R}^\mathbb{R}$
\item The set of differentiable real-valued functions $f$ on the interval $(0,3)$ such that $f'(2)=b$ is a subspace of $\mathbb{R}^{(0,3)}$ if and only if $b=0$.
\item The set of all sequences of complex numbers with limit $0$ is a subspace of $\mathbb{C}^\infty$.
\end{enumerate}
\begin{proof}[Answer]
\begin{enumerate}[label=(\alph*)]
\item Denote $A=\{(x_1,x_2,x_3,x_4)\in\mathbb{F}^4: x_3=5x_4+b\}$.

If $b=0$, then $A$ is a subspace by a verification similar to Exercise 1.C.2 (d).

On the other hand, if $A$ is a subspace, then $0\in A$. It follows that
\[
0=5*0+b=b.
\]
\item Denote $B=\{f\in\mathbb{R}^{[0,1]}:f \text{ is continuous}\}$.

We can verify that $0\in B$ because the constant function $f=0$ is continuous.

Suppose that $u,w\in B$, we can verify that $(u+w)\in B$ because the addition of two complex continuous functions is continuous by Theorem~4.9 of \cite{rudin1976principles}.

Suppose that $a\in\mathbb{F}$ and $u\in B$. Note that the constant function $f=a$ is continuous. We can verify that $au\in B$ because the multiplication of two continuous complex functions is continuous by Theorem~4.9 of \cite{rudin1976principles}.
\item Denote $C=\{f\in\mathbb{R}^{\mathbb{R}}:f \text{ is differentiable}\}$.

We can verify that $0\in C$ because the constant function $f=0$ is differentiable.

Suppose that $u,w\in C$, we can verify that $(u+w)\in C$ because the addition of two differentiable real functions is differentiable by Theorem~5.3 of \cite{rudin1976principles}.

Suppose that $a\in\mathbb{R}$ and $u\in C$. Note that the constant function $a(x)=a$ is differentiable. We can verify that $au\in C$ because the multiplication of two differentiable real functions is differentiable by Theorem~5.3 of \cite{rudin1976principles}.
\item Denote $D=\{f\in\mathbb{R}^{(0,3)}:f \text{ is differentiable and }f'(2)=b\}$.

Suppose that $b=0$. We claim that $D$ is a subspace of $\mathbb{R}^{(0,3)}$ with the following verifications.
\begin{enumerate}[label=(\roman*)]
\item $0\in D$ because the constant function $0(x)=0$ is differentiable and $0'(2)=0$.

We first verify that $W$ is a subspace of $\mathbb{F}^5$ by 1.34 of \cite{axler2014linear}.

\begin{enumerate}
\item We can show that $0\in W$ by choosing $x=0,y=0,z=0$ in $(0,0,x,y,z)$.
\item Suppose that $u,w\in W$ where \[u=(0,0,x_u,y_u,z_u), w=(0,0,x_w,y_w,z_w).\] Then $u+w=(0,0,x_u+x_w,y_u+y_w,z_u+z_w)$ shows that $u+w$ is in $W$.
\item Pick $a\in\mathbb{F}$ and $u\in W$ where $u=(0,0,x_u,y_u,z_u)$. Then \[au=(0,0,ax_u,ay_u,az_u)\] shows that $au$ is in $W$.
\end{enumerate}

We then verify that $\mathbb{F}^5=U\oplus W$ by 1.45 of \cite{axler2014linear}.

Pick $v\in U\cap W$. Then there exists $x,y,a,b,c\in\mathbb{F}$ such that $v=(x,y,x+y,x-y,2x)=(0,0,a,b,c)$, which implies that $x=0, y=0, a=0, b=0, c=0$, i.e. $v=0$. Therefore $U+W$ is a direct sum.
\end{proof}
