\section{Exercise 1.A.1}
Suppose $a$ and $b$ are real numbers, not both $0$. Find real numbers $c$ and $d$ such that
\[
1/(a+bi)=c+di
\]
\begin{proof}[Answer]
Since at least one of the real numbers $a, b$ is different from $0$, we have $a^2+b^2\ne 0$. We can then let
\[
c=\frac{a}{a^2+b^2},
\]
\[
d=\frac{-b}{a^2+b^2}.
\]
Then
\begin{equation*}
\begin{split}
&\left(a+bi\right)\left(\frac{a}{a^2+b^2}-\frac{b}{a^2+b^2}i\right)\\
&=\left(a\frac{a}{a^2+b^2}-b\left(-\frac{b}{a^2+b^2}\right)\right) + \left(a\left(-\frac{b}{a^2+b^2}\right)+b\frac{a}{a^2+b^2}\right)i\\
&=1 + 0i\\
&=1.
\end{split}
\end{equation*}
\end{proof}

\section{Exercise 1.A.2}
Show that
\[
\frac{-1+\sqrt{3}i}{2}
\]
is a cube root of $1$ (meaning that its cube equals $1$).
\begin{proof}[Answer]
\begin{equation*}
\begin{split}
&\frac{-1+\sqrt{3}i}{2}*\frac{-1+\sqrt{3}i}{2}*\frac{-1+\sqrt{3}i}{2}\\
&=\frac{-2-2\sqrt{3}i}{4}*\frac{-1+\sqrt{3}i}{2}\\
&=\frac{(-2)(-1)-(-2\sqrt{3})(\sqrt{3})+(-2\sqrt{3}+(-2\sqrt{3})(-1))i}{8}\\
&=1
\end{split}
\end{equation*}
\end{proof}

Find two distinct square roots of $i$.
\begin{proof}[Answer]

Thus the following two numbers are two distinct square roots of $i$:

\end{proof}

\section{Exercise 1.A.4}
Show that $\alpha+\beta=\beta+\alpha$ for all $\alpha, \beta\in\mathbb{C}$.
\begin{proof}
Suppose $\alpha=a+bi$ and $\beta=c+di$, where $a,b,c,d\in\mathbb{R}$. Then the definition of addition of complex numbers shows that
\[
\alpha+\beta=(a+c)+(b+d)i
\]
and
\[
\beta+\alpha=(c+a)+(d+b)i.
\]
The equations above and the commutativity of addition of real numbers show that $\alpha+\beta=\beta+\alpha$.
\end{proof}

\section{Exercise 1.A.5}
Show that $(\alpha+\beta)+\lambda=\alpha+(\beta+\lambda)$ for all $\alpha,\beta,\lambda\in\mathbb{C}$.
\begin{proof}
Suppose $\alpha=a+bi$, $\beta=c+di$ and $\lambda=e+fi$, where $a,b,c,d,e,f\in\mathbb{R}$. Then the definition of addition of complex numbers shows that
\begin{equation*}
\begin{split}
(\alpha+\beta)+\lambda&=((a+c)+(b+d)i)+e+fi\\
&=((a+c)+e)+((b+d)+f)i
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
\alpha+(\beta+\lambda)&=a+bi+((c+e)+(d+f)i)\\
&=(a+(c+e))+(b+(d+f))i.
\end{split}
\end{equation*}
The equations above and the associativity of addition of real numbers show that $(\alpha+\beta)+\lambda=\alpha+(\beta+\lambda)$.
\end{proof}

\section{Exercise 1.A.6}
Show that $(\alpha\beta)\lambda=\alpha(\beta\lambda)$ for all $\alpha,\beta,\lambda\in\mathbb{C}$.\begin{proof}
Suppose $\alpha=a+bi$, $\beta=c+di$ and $\lambda=e+fi$, where $a,b,c,d,e,f\in\mathbb{R}$. Then the definition of multiplication of complex numbers shows that
\begin{equation*}
\begin{split}
(\alpha\beta)\lambda&=((ac-bd)+(ad+bc)i)(e+fi)\\
&=((ac-bd)e-(ad+bc)f)+((ac-bd)f+(ad+bc)e)i\\
&=(ace-bde-adf-bcf)+(acf-bdf+ade+bce)i
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
\alpha(\beta\lambda)&=(a+bi)((ce-df)+(cf+de)i)\\
&=(a(ce-df)-b(cf+de))+(a(cf+de)+b(ce-df))i\\
&=(ace-adf-bcf-bde)+(acf+ade+bce-bdf)i.
\end{split}
\end{equation*}
The equations above and the commutativity of addition of real numbers show that $(\alpha\beta)\lambda=\alpha(\beta\lambda)$.
\end{proof}

\section{Exercise 1.A.7: Existence and Uniqueness of Additive Inverse in $\mathbb{C}$}
Show that for every $\alpha\in\mathbb{C}$, there exists a unique $\beta\in\mathbb{C}$ such that $\alpha+\beta=0$.
\begin{proof}
Suppose $\alpha=a+bi$ where $a,b\in\mathbb{R}$, then $\beta=-a-bi$ satisfies $\alpha+\beta=0$.

Suppose that there exists another $\beta'\in\mathbb{C}$ such that $\alpha+\beta'=0$. Then
\begin{equation*}
\begin{split}
\beta&=\beta+(\alpha+\beta')\\
&=(\beta+\alpha)+\beta'\\
&=\beta'.
\end{split}
\end{equation*}

The equations above show that $\beta'=\beta$.
\end{proof}

\section{Exercise 1.A.8: Existence and Uniqueness of Multiplicative Inverse in $\mathbb{C}$}
Show that for every $\alpha\in\mathbb{C}$ with $\alpha\ne 0$, there exists a unique $\beta\in\mathbb{C}$ such that $\alpha\beta=1$.
\begin{proof}
The existence of $\beta$ has been shown in Exercise 1.A.1.

Suppose that there exists another $\beta'\in\mathbb{C}$ such that and $\alpha\beta'=1$. Then
\begin{equation*}
\begin{split}
\beta&=\beta(\alpha\beta')\\
&=(\beta\alpha)\beta'\\
&=\beta'.
\end{split}
\end{equation*}

The equations above show that $\beta'=\beta$.
\end{proof}

\section{Exercise 1.A.9}
Show that $\lambda(\alpha+\beta)=\lambda\alpha+\lambda\beta$ for all $\lambda,\alpha,\beta\in\mathbb{C}$.
\begin{proof}
Suppose $\alpha=a+bi$, $\beta=c+di$ and $\lambda=e+fi$, where $a,b,c,d,e,f\in\mathbb{R}$. Then the definition of multiplication and addition of complex numbers shows that
\begin{equation*}
\begin{split}
\lambda(\alpha+\beta)&=(e+fi)((a+c)+(b+d)i)\\
&=(e(a+c)-f(b+d))+(e(b+d)+f(a+c))i\\
&=(ea+ec-fb-fd)+(eb+ed+fa+fc)i
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
\lambda\alpha+\lambda\beta&=(e+fi)(a+bi)+(e+fi)(c+di)\\
&=((ea-fb)+(eb+fa)i)+((ec-fd)+(ed+fc)i)\\
&=(ea-fb+ec-fd)+(eb+fa+ed+fc)i.
\end{split}
\end{equation*}

The equations above and the commutativity of addition of real numbers show that $\lambda(\alpha+\beta)=\lambda\alpha+\lambda\beta$.
\end{proof}

\section{Exercise 1.A.11: A Scalar Multiplication Example in $\mathbb{C}^3$}
Explain why there does not exist $\lambda\in\mathbb{C}$ such that
\[
\lambda(2-3i,5+4i,-6+7i)=(12-5i,7+22i,-32-9i).
\]
\begin{proof}
If there exists such $\lambda$ then the definition of scalar multiplication in $\mathbb{C}^3$ shows that,
\[
\lambda(2-3i)=12-5i.
\]

Thus
\[
\lambda=\frac{12-5i}{2-3i}=3+2i.
\]

But
\begin{equation*}
\begin{split}
\lambda(2-3i,5+4i,-6+7i)&=(3+2i)(2-3i,5+4i,-6+7i)\\
&=(12-5i,7+22i,-32+9i)\\
&\ne(12-5i,7+22i,-32-9i).
\end{split}
\end{equation*}

Therefore $\lambda$ does not exist.
\end{proof}

\section{Exercise 1.A.12: Associativity of Vector Addition in $\mathbb{F}^n$}
Show that $(x+y)+z=x+(y+z)$ for all $x,y,z\in\mathbb{F}^n$.
\begin{proof}
Suppose $x=(x_1,\dots,x_n)$, $y=(y_1,\dots,y_n)$ and $z=(z_1,\dots,z_n)$ where $x_1,\dots,x_n,y_1,\dots,y_n,z_1,\dots,z_n\in\mathbb{F}$. Then the definition of addition in $\mathbb{F}^n$ shows that
\begin{equation*}
\begin{split}
(x+y)+z&=(x_1+y_1,\dots,x_n+y_n)+(z_1,\dots,z_n)\\
&=((x_1+y_1)+z_1,\dots,(x_n+y_n)+z_n)
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
x+(y+z)&=(x_1,\dots,x_n)+(y_1+z_1,\dots,y_n+z_n)\\
&=(x_1+(y_1+z_1),\dots,x_n+(y_n+z_n))
\end{split}
\end{equation*}

The equations above and the associativity of addition in $\mathbb{F}$ show that $(x+y)+z=x+(y+z)$.
\end{proof}

\section{Exercise 1.A.13: Associativity of Scalar Multiplication in $\mathbb{F}^n$}
Show that $(ab)x=a(bx)$ for all $x\in\mathbb{F}^n$ and all $a,b\in\mathbb{F}$.
\begin{proof}
Suppose $x=(x_1,\dots,x_n)$ where $x_1,\dots,x_n\in\mathbb{F}$. Then the definition of scalar multiplication in $\mathbb{F}^n$ shows that
\begin{equation*}
\begin{split}
(ab)x&=((ab)x_1,\dots,(ab)x_n)
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
a(bx)&=a(bx_1,\dots,bx_n)\\
&=(a(bx_1),\dots,a(x_n))
\end{split}
\end{equation*}

The equations above and the associativity of multiplication in $\mathbb{F}$ show that $(ab)x=a(bx)$.
\end{proof}

\section{Exercise 1.A.14: Scalar Multiplication by $1$ in $\mathbb{F}^n$}
Show that $1x=x$ for all $x\in\mathbb{F}^n$.
\begin{proof}
Suppose $x=(x_1,\dots,x_n)$ where $x_1,\dots,x_n\in\mathbb{F}$. Then the definition of scalar multiplication in $\mathbb{F}^n$ and the fact that $1$ is the multiplicative identity of $\mathbb{F}$ shows that
\begin{equation*}
\begin{split}
1x&=(1x_1,\dots,1x_n)\\
&=(x_1,\dots,x_n)\\
&=x.
\end{split}
\end{equation*}
\end{proof}

\section{Exercise 1.A.15: Distributing Vector Additions with Scalar Multiplications in $\mathbb{F}^n$}
Show that $\lambda(x+y)=\lambda x+\lambda y$ for all $\lambda\in\mathbb{F}$ and $x,y\in\mathbb{F}^n$.
\begin{proof}
Suppose $x=(x_1,\dots,x_n)$ and $y=(y_1,\dots,y_n)$ where $x_1,\dots,x_n,y_1,\dots,y_n\in\mathbb{F}$. Then the definition of addition and scalar multiplication in $\mathbb{F}^n$ shows that
\begin{equation*}
\begin{split}
\lambda(x+y)&=\lambda(x_1+y_1,\dots,x_n+y_n)\\
&=(\lambda(x_1+y_1),\dots,\lambda(x_n+y_n))
\end{split}
\end{equation*}
and
\begin{equation*}
\begin{split}
\lambda x+\lambda y&=(\lambda x_1,\dots,\lambda x_n) + (\lambda y_1,\dots,\lambda y_n)\\
&=(\lambda x_1+\lambda y_1,\dots,\lambda x_n+\lambda y_n)
\end{split}
\end{equation*}

The equations above and the distributive property of $\mathbb{F}$ show that $\lambda(x+y)=\lambda x+\lambda y$.
\end{proof}

\section{Exercise 1.A.16: Distributing Scalar Additions with Scalar Multiplications in $\mathbb{F}^n$}
Show that $(a+b)x=ax+bx$ for all $a,b\in\mathbb{F}$ and all $x\in\mathbb{F}^n$.
\begin{proof}
Suppose $x=(x_1,\dots,x_n)$ where $x_1,\dots,x_n\in\mathbb{F}$. Then the definition of scalar multiplication and addition in $\mathbb{F}^n$ shows that
\[
(a+b)x=((a+b)x_1,\dots,(a+b)x_n)
\]
and
\begin{equation*}
\begin{split}
ax+bx&=(ax_1,\dots,ax_n)+(bx_1,\dots,bx_n)\\
&=(ax_1+bx_1,\dots,ax_n+bx_n).
\end{split}
\end{equation*}

The equations above and the distributive property of $\mathbb{F}$ show that $(a+b)x=ax+bx$.
\end{proof}

\section{Exercise 1.B.1}
Prove that $-(-v)=v$ for every $v\in V$.
\begin{proof}
\begin{equation*}
\begin{split}
-(-v)&=-(-v)+0\\
&=-(-v)+((-v)+v)\\
&=(-(-v)+(-v))+v\\
&=0+v\\
&=v
\end{split}
\end{equation*}
\end{proof}

\section{Exercise 1.B.2}
Suppose $a\in\mathbb{F}, v\in V$, and $av=0$. Prove that $a=0$ or $v=0$.
\begin{proof}
Suppose that $v=(v_1,\dots,v_n)$ where $v_1,\dots,v_n\in\mathbb{F}$. Then by the definition of scalar multiplication
\[
av=(av_1,\dots,av_n).
\]

For the purpose of arriving at a contradiction, assume that both $a\ne 0$ and $v\ne 0$.

Now $v\ne 0$ implies that there exists some $i\in\{1,\dots,n\}$ such that $v_i\ne 0$. Thus $av_i\ne 0$, which leads to $av\ne 0$ contradicting with the premise that $av=0$.

Therefore $a=0$ or $v=0$.
\end{proof}

\section{Exercise 1.B.3}
Suppose $v,w\in V$. Explain why there exists a unique $x\in V$ such that $v+3x=w$.
\begin{proof}
We claim that the following $x\in V$ satisfies $v+3x=w$:
\[
x=\frac{1}{3}(w-v).
\]

We can verify it by showing that,
\begin{equation*}
\begin{split}
v+3x&=v+3\times\frac{1}{3}(w-v)\\
&=v+w-v\\
&=w.
\end{split}
\end{equation*}

Suppose that there exists another $x'\in V$ that also satisfies $v+3x'=w$, then
\begin{equation*}
\begin{split}
0&=w-w\\
&=(v+3x)-(v+3x')\\
&=v+3x-v-3x'\\
&=3x-3x'.
\end{split}
\end{equation*}

Thus
\begin{equation*}
\begin{split}
x'&=x'+\frac{1}{3}\times0\\
&=x'+\frac{1}{3}(3x-3x')\\
&=x'+x-x'\\
&=x.
\end{split}
\end{equation*}
\end{proof}

\section{Exercise 1.B.5}
Show that in the definition of a vector space (1.19), the additive inverse condition can be replaced with the condition that
\[
0v=0, \forall v\in V.
\]

Here the $0$ on the left side is the number $0$, and the $0$ on the right side is the additive identity of $V$. 
\begin{proof}
We will show that the new condition is equal to the original condition, which states that for every $v\in V$, there exists $w\in V$ such that $v+w=0$. That the old condition implies the new condition has been shown in Theorem 1.29 ``The number $0$ times a vector''. Therefore it suffices to show that the new condition implies the old condition.

Suppose that the new condition holds.

Pick $v\in V$, and let $w=(-1)v$. It follows that
\begin{equation*}
\begin{split}
v+w&=1v+(-1)v\\
&=(1-1)v\\
&=0v\\
&=0.
\end{split}
\end{equation*}

Therefore the old condition holds.
\end{proof}
